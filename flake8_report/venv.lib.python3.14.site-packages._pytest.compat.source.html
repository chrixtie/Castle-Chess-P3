195 def ascii_escaped(val: bytes | str) -> str:
196 r"""If val is pure ASCII, return it as an str, otherwise, escape
197 bytes objects into a sequence of escaped bytes:
198  
199 b'\xc3\xb4\xc5\xd6' -> r'\xc3\xb4\xc5\xd6'
200  
201 and escapes strings into a sequence of escaped unicode ids, e.g.:
202  
203 r'4\nV\U00043efa\x0eMXWB\x1e\u3028\u15fd\xcd\U0007d944'
204  
205 Note:
206 The obvious "v.decode('unicode-escape')" will return
207 valid UTF-8 unicode if it finds them in bytes, but we
208 want to return escaped bytes for any byte, even if they match
209 a UTF-8 string.
210 """
211 if isinstance(val, bytes):
212 ret = val.decode("ascii", "backslashreplace")
213 else:
214 ret = val.encode("unicode_escape").decode("ascii")
215 return ret.translate(_non_printable_ascii_translate_table)

260 def get_user_id() -> int | None:
261 """Return the current process's real user id or None if it could not be
262 determined.
263  
264 :return: The user id or None if it could not be determined.
265 """
266 # mypy follows the version and platform checking expectation of PEP 484:
267 # https://mypy.readthedocs.io/en/stable/common_issues.html?highlight=platform#python-version-and-system-platform-checks
268 # Containment checks are too complex for mypy v1.5.0 and cause failure.
269 if sys.platform == "win32" or sys.platform == "emscripten":
270 # win32 does not have a getuid() function.
271 # Emscripten has a return 0 stub.
272 return None
273 else:
274 # On other platforms, a return value of -1 is assumed to indicate that
275 # the current process's real user id could not be determined.
276 ERROR = -1
277 uid = os.getuid()
278 return uid if uid != ERROR else None
